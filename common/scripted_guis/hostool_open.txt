	bt_open= {
		context_type = player_context

		window_name = "bt_open_container"
		visible = {
			
		}

				else = {
					ROOT = { clr_country_flag = hosttool_enabled }
				}
			}
		}
		triggers = {
			bt_open_click_enabled = {
				always = yes
			}
		}
	}
